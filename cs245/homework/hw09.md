public class HW09DataPoints {
	int CAP = 100;
	int[] data = new int[CAP];

	public HW09DataPoints(int[] initialData) {
		data = initialData;
	}

	public int find(int q) {
		for (int i = 0; i < data.length; i++) {
			if (data[i] == q) {
				return i;
			}
		}
		return -1;
	}

	public double findrelevance(int q) {
		int count = 0;
		for (int i = 0; i < data.length; i++) {
			if (data[i] == q) {
				count++;
			}
		}
		return (double)count/data.length;
	}

	public int findlast(int q) {
		int lastindex = -1;
		for (int i = 0; i < data.length; i++) {
			if (data[i] == q) {
				lastindex = i;
			}
		}
		return lastindex;
	}

	public int getSize() {
		return data.length;
	}

	public int getCapacity() {
		return CAP;
	}

	/* Implements a bubble sort */
	public int sort() {
		for (int i = data.length-1; i > 0; i--) {
			for (int j = 0; j < i; j++) {
				if (data[j] > data[j+1]) {
					int t = data[j];
					data[j] = data[j+1];
					data[j+1] = t;
				}
			}
		}
		return data[data.length-1];
	}

	public int findBinary(int q) {
		int bot = 0;
		int top = data.length - 1;
		int mid = 0;
		int win = 0;
		
		while (top >= bot) {
			win = bot + top; // establish win
			mid = win / 2;  
			if (data[mid] == q) {
				return mid;
			} else if (q > data[mid]) {
				bot = mid + 1;  // shift window right
			} else {
				top = mid - 1;  // shift window left
			}
		}
		return -1;
	}


	public void print() {
		for (int i: data) {
			System.out.print(i + ", ");
		}
		System.out.println();
	}
